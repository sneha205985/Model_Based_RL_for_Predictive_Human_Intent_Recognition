\subsection{Gaussian Process Learning Guarantees}

\subsubsection{Convergence and Calibration}

\begin{theorem}[GP Learning Performance Summary]
\label{thm:gp_summary}
The Bayesian Gaussian Process for human behavior learning provides:

\textbf{Convergence Guarantees:}
\begin{align}
\lim_{n \to \infty} \hat{f}_n(x) &= f^*(x) \quad \text{almost surely} \\
\E[|\hat{f}_n(x) - f^*(x)|^2] &= \mathcal{O}(n^{-\frac{2s}{2s + d}})
\end{align}

\textbf{Uncertainty Calibration:}
$$\Pr[f^*(x) \in CI_{1-\alpha}(x)] = 1 - \alpha \quad \text{(exact under correct specification)}$$

\textbf{Information Gain Bounds:}
$$\gamma_n \leq \frac{1}{2}\log\det(I + \sigma_n^{-2}K_n) = \mathcal{O}((\log n)^{d+1})$$

\textbf{Real-time Implementation:}
- Exact GP: $\mathcal{O}(n^3)$ preprocessing, $\mathcal{O}(n^2)$ per prediction
- Sparse GP: $\mathcal{O}(m^3)$ preprocessing, $\mathcal{O}(m^2)$ per prediction with $m \ll n$
- Memory requirement: $\mathcal{O}(n^2)$ for exact, $\mathcal{O}(nm + m^2)$ for sparse
\end{theorem}

\subsubsection{Practical Performance Bounds}

- State dimension: $d = 10-20$

\subsection{MPC Control Guarantees}

\subsubsection{Stability and Robustness}

\begin{theorem}[MPC Control Performance Summary]
\label{thm:mpc_summary}
The Model Predictive Controller provides:

\textbf{Stability Guarantees:}
\begin{align}
\lim_{t \to \infty} x(t) &= x_s \quad \text{(asymptotic stability)} \\
\norm{x(t) - x_s} &\leq Ce^{-\lambda t}\norm{x(0) - x_s} \quad \text{(exponential convergence)}
\end{align}

\textbf{Robustness to GP Uncertainty:}
$$\limsup_{t \to \infty} \norm{x(t) - x_s} \leq \rho \cdot \delta_{GP}$$
where $\delta_{GP}$ is the GP prediction error bound.

\textbf{Input-to-State Stability:}
$$\norm{x(t) - x_s} \leq \beta(\norm{x(0) - x_s}, t) + \gamma\left(\sup_{\tau \leq t} \norm{w(\tau)}\right)$$

\textbf{Performance Bounds:}
\begin{align}
\sum_{t=0}^{T-1} \ell(x(t), u(t)) &\leq V_N^*(x(0)) \quad \text{(finite horizon)} \\
\sum_{t=0}^{\infty} \ell(x(t), u(t)) &< \infty \quad \text{(infinite horizon)}
\end{align}
\end{theorem>

\subsubsection{Real-time Implementation Bounds}

\begin{corollary}[MPC Real-time Performance]

For industrial HRI applications:
- Control frequency: 100-500 Hz

\subsection{Reinforcement Learning Guarantees}

\subsubsection{Regret and Sample Complexity}

\begin{theorem}[RL Performance Summary]
\label{thm:rl_summary}
The Bayesian RL agent achieves:

\textbf{Regret Bounds:}
\begin{align}
\E[\text{BayesRegret}(T)] &\leq C\sqrt{T H^3 |\mathcal{S}|^2 |\mathcal{A}| \log T} \\
\Pr[\text{Regret}(T) \leq \epsilon] &\geq 1 - \delta \quad \text{with appropriate } \epsilon(\delta)
\end{align}

\textbf{Sample Complexity:}
$$T_{\epsilon,\delta} = \mathcal{O}\left(\frac{H^4 |\mathcal{S}|^2 |\mathcal{A}| \log(|\mathcal{S}||\mathcal{A}|H/\delta)}{\epsilon^2}\right)$$

\textbf{Convergence Guarantees:}
\begin{align}
\lim_{t \to \infty} \pi_t &= \pi^* \quad \text{almost surely} \\
\E[\norm{V^{\pi_t} - V^{\pi^*}}_\infty] &= \mathcal{O}\left(\sqrt{\frac{\gamma_t}{t}}\right)
\end{align}

\textbf{Multi-Agent Performance:}
- Individual regret: $\mathcal{O}(\sqrt{T/N})$ per agent with $N$ agents
- Collective learning: $\mathcal{O}(\sqrt{NT})$ improvement
- Communication complexity: $\mathcal{O}(\sqrt{T}/\epsilon^2)$ for federated learning
\end{theorem}

\subsubsection{Probabilistic Safety Bounds}

\begin{theorem}[Integrated Safety Summary]
\label{thm:safety_summary}
The complete integrated system provides:

\textbf{Probabilistic Safety:}
$$\Pr[\forall t \in [0,T], x(t) \in \mathcal{S}] \geq 1 - \delta_{\text{total}}$$
where $\delta_{\text{total}} \leq \delta_{\text{GP}} + \delta_{\text{MPC}} + \delta_{\text{RL}} + \delta_{\text{interaction}}$.

\textbf{Time-Improving Safety:}
$$\Pr[x(t) \in \mathcal{S}] \geq 1 - \delta_0 e^{-\lambda t} - \delta_{\infty}$$

\textbf{Barrier Function Safety:}
If $B(x(t), t) \geq 0$ initially, then $B(x(t'), t') \geq 0$ for all $t' > t$.

\textbf{Multi-Agent Safety:}
$$\Pr[\text{collision-free}] \geq 1 - N(N-1) \delta_{\text{pairwise}}$$
for $N$ agents with pairwise collision probability $\delta_{\text{pairwise}}$.

\textbf{Runtime Verification:}
- Safety monitor computation: $<1$ ms
- Predictive safety horizon: up to 10 seconds with 95\% accuracy
- Fault tolerance: up to $\lfloor(N-1)/2\rfloor$ Byzantine failures
\end{theorem}
